Integrating Components into a Unified Application

In this session, we’ll connect all the components of the HeartLens app to create a fully functional prototype. Specifically, we’ll:

  • Add a "Start Recording" button that triggers the camera feed and begins recording data.

  • Use the selected signal mode from the SignalConfig component to simulate data.

  • Display the recorded data in both the HeartRateMonitor and VisualizationPanel.

By the end of this session, you’ll have an integrated app where all components work together seamlessly.

1. Centralized State Management in app/page.tsx

The parent component will reside in app/page.tsx, which is the default entry point for the App Router in Next.js. This component will manage shared state and pass it down to child components.

Step 1: Update app/page.tsx

Replace the contents of app/page.tsx with the following code:

"use client";
import { useState } from "react";
import CameraFeed from "./components/CameraFeed";
import SignalConfig from "./components/SignalConfig";
import HeartRateMonitor from "./components/HeartRateMonitor";
import VisualizationPanel from "./components/VisualizationPanel";

export default function Home() {
  // Shared State
  const [isRecording, setIsRecording] = useState(false);
  const [signalMode, setSignalMode] = useState("Green Only");
  const [ppgData, setPpgData] = useState<number[]>([]);
  const [intervalId, setIntervalId] = useState<NodeJS.Timeout | null>(null);

  // Simulate Data Generation
  const startRecording = () => {
    setIsRecording(true);

    const newInterval = setInterval(() => {
      const newData = Math.random() * 100; // Simulated PPG value
      setPpgData((prevData) => [...prevData.slice(-20), newData]); // Keep last 20 points
    }, 500);

    setIntervalId(newInterval); // Save the interval ID
  };

  const stopRecording = () => {
    setIsRecording(false);

    if (intervalId) {
      clearInterval(intervalId); // Clear the interval
      setIntervalId(null); // Reset the interval ID
    }

    setPpgData([]); // Clear recorded data
  };

  return (
    <div>
      <h1>HeartLens App</h1>

      {/* Signal Configuration */}
      <SignalConfig signalMode={signalMode} setSignalMode={setSignalMode} />

      {/* Start/Stop Recording Button */}
      <div>
        <button onClick={isRecording ? stopRecording : startRecording}>
          {isRecording ? "⏹️ Stop Recording" : "▶️ Start Recording"}
        </button>
      </div>

      {/* Camera Feed */}
      <CameraFeed isRecording={isRecording} />

      {/* Real-Time Visualization */}
      <VisualizationPanel ppgData={ppgData} />

      {/* Heart Rate Monitor */}
      <HeartRateMonitor ppgData={ppgData} />
    </div>
  );
}
```

Explanation of the Code

  1. Shared State :

    • isRecording: Tracks whether recording is active (true or false).

    • signalMode: Stores the selected signal mode (e.g., "Green Only").

    • ppgData: Stores the simulated PPG data as an array of numbers.

  2. Start/Stop Recording :

    • The startRecording function generates random PPG data every 500ms and updates the ppgData state. It also sets isRecording to true.

    • The stopRecording function clears the interval, resets isRecording to false, and clears the ppgData.

  3. Passing Props :

    • SignalConfig: Receives signalMode and setSignalMode as props to manage the selected mode.

    • CameraFeed: Receives isRecording to control the camera feed.

    • VisualizationPanel and HeartRateMonitor: Receive ppgData to display the recorded data.
```

2.2 Update CameraFeed Component

Step 1: Modify CameraFeed.tsx

Update the CameraFeed component to accept isRecording as a prop:

"use client";
import { useState, useEffect } from "react";

interface CameraFeedProps {
  isRecording: boolean;
}

export default function CameraFeed({ isRecording }: CameraFeedProps) {
  const [stream, setStream] = useState<MediaStream | null>(null);

  const startCamera = async () => {
    try {
      const newStream = await navigator.mediaDevices.getUserMedia({
        video: { facingMode: "environment" },
      });
      const videoElement = document.getElementById("camera-feed") as HTMLVideoElement;
      if (videoElement) {
        videoElement.srcObject = newStream;
        videoElement.play();
      }
      setStream(newStream);
    } catch (err) {
      console.error("Error accessing camera:", err);
    }
  };

  const stopCamera = () => {
    if (stream) {
      stream.getTracks().forEach((track) => track.stop());
      setStream(null);
    }
  };

  useEffect(() => {
    if (isRecording) {
      startCamera();
    } else {
      stopCamera();
    }
  }, [isRecording]);

  return (
    <div>
      <h3>Camera Feed</h3>
      <video id="camera-feed" width="320" height="240" autoPlay playsInline></video>
    </div>
  );
}

2.3 Update HeartRateMonitor Component

Step 1: Modify HeartRateMonitor.tsx

Update the HeartRateMonitor component to accept ppgData as a prop and calculate the heart rate based on the latest PPG value:

"use client";
import { useState, useEffect } from "react";

interface HeartRateMonitorProps {
  ppgData: number[];
}

export default function HeartRateMonitor({ ppgData }: HeartRateMonitorProps) {
  const [heartRate, setHeartRate] = useState(72);

  // Calculate heart rate based on the latest PPG data
  useEffect(() => {
    if (ppgData.length > 0) {
      const latestPpgValue = ppgData[ppgData.length - 1];
      const newHeartRate = Math.floor(latestPpgValue); // Simulate heart rate calculation
      setHeartRate(newHeartRate);
    }
  }, [ppgData]);

  return (
    <div>
      <h2>Heart Rate Monitor</h2>
      <p>Current Heart Rate: {heartRate} BPM</p>
    </div>
  );
}

2.4 Update VisualizationPanel Component

Step 1: Modify VisualizationPanel.tsx

"use client";
import { Chart as ChartJS, CategoryScale, LinearScale, PointElement, LineElement, Title, Tooltip, Legend } from "chart.js";
import { Line } from "react-chartjs-2";

// Register the required components
ChartJS.register(CategoryScale, LinearScale, PointElement, LineElement, Title, Tooltip, Legend);

interface VisualizationPanelProps {
  ppgData: number[];
}

export default function VisualizationPanel({ ppgData }: VisualizationPanelProps) {
  // Chart.js configuration
  const chartData = {
    labels: Array.from({ length: ppgData.length }, (_, i) => i.toString()),
    datasets: [
      {
        label: "PPG Signal",
        data: ppgData,
        borderColor: "rgb(75, 192, 192)",
        tension: 0.4,
        fill: true,
        backgroundColor: "rgba(75, 192, 192, 0.2)",
        pointRadius: 0,
      },
    ],
  };

  const chartOptions = {
    responsive: true,
    scales: {
      y: {
        beginAtZero: false,
      },
    },
    animation: {
      duration: 0, // Disable animation for better performance
    },
  };

  return (
    <div>
      <h3>Real-Time Visualization</h3>
      <Line data={chartData} options={chartOptions} />
    </div>
  );
}
